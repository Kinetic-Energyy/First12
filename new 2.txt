Write a C++ program  using template exchange ( ) that accepts two arguments of generic type and add their contents.


#include<iostream>
using namespace std;

template <typename T>
T exchange(T a, T b)
{
	T temp;
	temp = a;
	a = b;
	b = temp;
	return a + b;
}

int main()
{
	int x = 5;
	int y = 10;
	
	// Exchange the contents of x and y
	cout<<"The sum after exchanging the contents of x and y is: "<< exchange(x, y) <<endl;
	
	float a = 5.0;
	float b = 10.0;
	
	// Exchange the contents of a and b
	cout<<"The sum after exchanging the contents of a and b is: "<< exchange(a, b) <<endl;
	
	return 0;
}
